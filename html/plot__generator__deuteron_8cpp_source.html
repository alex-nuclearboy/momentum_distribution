36void PlotGeneratorDeuteron::GenerateSinglePlot(
37 const std::string& model_name, const std::string& data_file,
38 const std::string& plot_file)
39{
40 std::ifstream in_file(data_file);
41 if (!in_file.is_open()) {
42 std::cerr << "Error: Failed to open the data file for reading: "
43 << data_file << std::endl;
44 return;
45 }
46
47 std::vector<double> momentum;
48 std::vector<double> density;
49 double p, d;
50
51 while (in_file >> p >> d) {
52 momentum.push_back(p);
53 density.push_back(d);
54 }
55 in_file.close();
56
57 // Prepare the TGraph
58 TGraph* graph = new TGraph(momentum.size(), &momentum[0], &density[0]);
59 std::string title = "Nucleon Momentum Distribution (" + model_name +
60 " potential);Momentum (GeV/c);Probability Density";
61 graph->SetTitle(title.c_str());
62 graph->SetMarkerStyle(20);
63 graph->SetMarkerSize(0.7);
64 graph->SetDrawOption("AP");
65
66 // Create a TCanvas and draw the TGraph
67 TCanvas* canvas = new TCanvas("c1", model_name.c_str(), 800, 600);
68 graph->Draw("APL");
69
70 // Save the plot
71 canvas->SaveAs(plot_file.c_str());
72 delete canvas; // Clean up
73}

80void PlotGeneratorDeuteron::GenerateCombinedPlot(
81 const nlohmann::json& models, const std::string& combined_plot_file)
82{
83 TMultiGraph *mg = new TMultiGraph();
84 TLegend *legend = new TLegend(0.7, 0.7, 0.9, 0.9);
85
86 for (const auto& model : models) {
87 std::string model_name = model["name"];
88 std::string data_file = "data/" + model_name
89 + "_momentum_distribution.txt";
90
91 std::ifstream in_file(data_file);
92 if (!in_file.is_open()) {
93 std::cerr << "Error: Failed to open the data file for reading: "
94 << data_file << std::endl;
95 continue; // Skip this model if the file can't be opened
96 }
97
98 std::vector<double> momentum;
99 std::vector<double> density;
100 double p, d;
101 while (in_file >> p >> d) {
102 momentum.push_back(p);
103 density.push_back(d);
104 }
105 in_file.close();
106
107 TGraph* graph = new TGraph(
108 momentum.size(), momentum.data(), density.data()
109 );
110 graph->SetTitle(model_name.c_str());
111 graph->SetLineWidth(2);
112 graph->SetLineStyle(1 + &model - &models[0]); // Different line style for each model
113 graph->SetLineColor(1 + &model - &models[0]); // Different line color for each model
114 mg->Add(graph);
115 legend->AddEntry(graph, (model_name + " potential").c_str(), "lp");
116 }
117
118 TCanvas *canvas = new TCanvas(
119 "c2", "Combined Fermi Momentum Distribution", 800, 600
120 );
121 mg->GetXaxis()->SetRangeUser(0., 0.4);
122 mg->GetYaxis()->SetRangeUser(0., 0.012);
123 mg->Draw("ALP");
124 std::string title = "Fermi Momentum Distribution;Momentum (GeV/c);"
125 "Probability Density (c/GeV)";
126 mg->SetTitle(title.c_str());
127 legend->Draw();
128
129 canvas->SaveAs(combined_plot_file.c_str());
130 delete canvas; // Clean up
131}

34{
35 int n = c.size();

38 for (int i = 0; i < n - 1; i++) {c[n - 1] -= c[i];}
39
40 int n2 = n - 3;
41 int n1 = n2 + 1;
42 int n0 = n1 + 1;
43
44 double sum1 = 0., sum2 = 0., sum3 = 0.;
45
46 // Summation for the normalization of d coefficients
47 for (int j = 0; j <= n - 3; ++j) {
48 sum1 += d[j] / m2[j];
49 sum2 += d[j];
50 sum3 += d[j] * m2[j];
51 }
52
53 // Loop to calculate the last three 'd' coefficients
54 for (int i = 0; i < 3; ++i) {
55 // Normalize the current d coefficient
56 d[n2] = -m2[n1] * m2[n0] * sum1 + (m2[n1] + m2[n0]) * sum2 - sum3;
57 d[n2] *= m2[n2] / ((m2[n0] - m2[n2]) * (m2[n1] - m2[n2]));
58
59 // Cycle the indices for the next iteration
60 std::swap(n0, n1);
61 std::swap(n1, n2);
62 }
63}
